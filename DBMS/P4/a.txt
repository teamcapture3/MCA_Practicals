CREATE TABLE EQUIPMENT (
ID VARCHAR2 (5) NOT NULL,
ITEM_NAME VARCHAR2 (20) NOT NULL,
CATEGORY VARCHAR2 (20) NOT NULL,
PRICE NUMBER NOT NULL,
QUANTITY_AVAIL NUMBER NOT NULL,
QUANTITY_ORDER NUMBER NOT NULL,
ORDER_DATE DATE NOT NULL,
DISPATCH_DATE DATE NOT NULL
);

DESC EQUIPMENT;

SELECT * FROM EQUIPMENT;

SELECT ID,ITEM_NAME,(QUANTITY_ORDER-QUANTITY_AVAIL) AS ORDERS_PENDING FROM EQUIPMENT;

SELECT ID,ITEM_NAME,ROUND(PRICE) FROM EQUIPMENT;

SELECT ID,ITEM_NAME,ROUND(PRICE,2) FROM EQUIPMENT;

SELECT ID,ITEM_NAME,SIGN(QUANTITY_ORDER-QUANTITY_AVAIL) AS ORDERS_PENDING FROM EQUIPMENT;

SELECT ID,ITEM_NAME AS OFFER_APPLICABLE FROM EQUIPMENT WHERE QUANTITY_ORDER=3*QUANTITY_AVAIL;

SELECT INITCAP(ITEM_NAME) FROM EQUIPMENT;

SELECT LPAD(ITEM_NAME,20,'**') FROM EQUIPMENT;

SELECT TO_CHAR(ADD_MONTHS(ORDER_DATE,3),'DD-MON-YYYY') "DESPATCH DATE"
FROM EQUIPMENT WHERE CATEGORY='HARDWARE';

SELECT ID,ITEM_NAME,TO_CHAR(TO_DATE(ORDER_DATE,'DD-MON-RR'),'YYYY')"ORDER"
FROM EQUIPMENT WHERE TO_CHAR(TO_DATE(ORDER_DATE,'DD-MON-RR'),'YYYY')='2020';
