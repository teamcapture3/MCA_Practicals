SET SERVEROUTPUT ON;

DECLARE
A NUMBER:=33;
B NUMBER:=31;
C NUMBER:=32;
BEGIN
DBMS_OUTPUT.PUT_LINE('A='||A||' B='||B||' C='||C);
IF A>B AND A>C
THEN
DBMS_OUTPUT.PUT_LINE('A IS GREATEST');
ELSE
IF B>A AND B>C
THEN
DBMS_OUTPUT.PUT_LINE('B IS GREATEST');	
ELSE
DBMS_OUTPUT.PUT_LINE('C IS GREATEST');
END IF;
END IF;
END;
/
