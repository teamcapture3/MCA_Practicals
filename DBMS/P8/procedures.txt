// CREATING PROCEDURES //
CREATE OR REPLACE PROCEDURE FINDMAX(X IN NUMBER,Y IN NUMBER,Z OUT NUMBER)
IS 
BEGIN
IF X>Y THEN
Z:=X;
ELSE
Z:=Y;
END IF;
END;
/

// CALLING PROCEDURES //
DECLARE 
A NUMBER;
B NUMBER;
C NUMBER;
BEGIN
A:=28;
B:=33;
FINDMAX(A,B,C);
DBMS_OUTPUT.PUT_LINE('MAXIMUM OF TWO NO IS : '||C);
END;
/

CREATE OR REPLACE PROCEDURE UPDATE_CUSTOMER(
P_USERID IN CUSTOMERS.ID%TYPE,
P_SALARY IN CUSTOMERS.SALARY%TYPE)
IS
BEGIN
UPDATE CUSTOMERS SET SALARY = P_SALARY WHERE ID = P_USERID;
COMMIT;
END;
