SET SERVEROUTPUT ON;

CREATE TABLE CUSTOMERS (
ID INT,
NAME VARCHAR2 (10),
SALARY INT
);

INSERT INTO CUSTOMERS VALUES (31, 'NINAD', 45000);

INSERT INTO CUSTOMERS VALUES (32, 'REIGNZ', 55000);

INSERT INTO CUSTOMERS VALUES (33, 'BLITZ', 60000);

DESC CUSTOMERS;

CREATE PACKAGE CUST_SALARY AS
PROCEDURE FIND_SALARY(C_ID CUSTOMERS.ID%TYPE);
END CUST_SALARY;
/

CREATE OR REPLACE PACKAGE BODY CUST_SALARY AS
PROCEDURE FIND_SALARY(C_ID CUSTOMERS.ID%TYPE) IS
C_SAL CUSTOMERS.SALARY%TYPE;
BEGIN
SELECT SALARY INTO C_SAL FROM CUSTOMERS
WHERE ID = C_ID;
DBMS_OUTPUT.PUT_LINE('SALARY IS: ' || C_SAL);
END FIND_SALARY;
END CUST_SALARY;
/

DECLARE
CODE CUSTOMERS.ID%TYPE:= &CC_ID;
BEGIN
CUST_SALARY.FIND_SALARY(CODE);
END;
/
