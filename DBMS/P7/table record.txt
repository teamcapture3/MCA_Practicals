set serveroutput on;

// Q1 //
DECLARE
T_EMP NUMBER;
BEGIN
SELECT COUNT(*)
INTO T_EMP
FROM EMPLOYEE;
DBMS_OUTPUT.PUT_LINE('TOTAL NUMBER OF EMPLOYEES: ' || TO_CHAR(T_EMP));
END;
/

// Q2 //
DECLARE
E_NAME EMPLOYEE.NAME%TYPE;
BEGIN
SELECT NAME
INTO E_NAME
FROM EMPLOYEE
WHERE ID=5;
IF SQL%FOUND THEN
DBMS_OUTPUT.PUT_LINE('EMPLOYEE NAME: ' || TO_CHAR(E_NAME));
ELSE
DBMS_OUTPUT.PUT_LINE('NO SUCH RECORD');
END IF;
END;
/


// Q3 //
DECLARE
E_ID NUMBER:=&E_ID;
E_NAME EMPLOYEE.NAME%TYPE;
BEGIN
SELECT NAME
INTO E_NAME
FROM EMPLOYEE
WHERE ID=E_ID;
DBMS_OUTPUT.PUT_LINE(E_NAME);
END;
/


// Q4 //
DECLARE
E_ID NUMBER:=&E_ID;
E_NAME EMPLOYEE.NAME%TYPE;
BEGIN
UPDATE EMPLOYEE
SET SALARY=30000
WHERE ID=E_ID AND SALARY<20000;
IF SQL%NOTFOUND THEN
DBMS_OUTPUT.PUT_LINE('UP TO DATE');
ELSE
DBMS_OUTPUT.PUT_LINE('UPDATE DONE: ' || TO_CHAR(E_NAME));
END IF;
END;
/
