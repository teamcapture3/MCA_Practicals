set serveroutput on;

// Q1 //
DECLARE
E_CITY EMPLOYEE.CITY%TYPE;
E_SALARY EMPLOYEE.SALARY%TYPE;
CURSOR EMP IS SELECT CITY,SUM(SALARY) FROM EMPLOYEE GROUP BY CITY;
BEGIN
OPEN EMP;
LOOP
FETCH EMP INTO E_CITY,E_SALARY;
EXIT WHEN EMP%NOTFOUND;
DBMS_OUTPUT.PUT_LINE(E_CITY || ' ' || E_SALARY);
END LOOP;
END;
/

// Q2 //
DECLARE
E_ID EMPLOYEE.ID%TYPE;
E_NAME EMPLOYEE.NAME%TYPE;
E_DOB EMPLOYEE.DOB%TYPE;
E_CITY EMPLOYEE.CITY%TYPE;
E_SALARY EMPLOYEE.SALARY%TYPE;
E_DNO EMPLOYEE.D_NO%TYPE;
CURSOR EMP IS SELECT * FROM EMPLOYEE WHERE SALARY>20000;
BEGIN
OPEN EMP;
LOOP
FETCH EMP INTO E_ID,E_NAME,E_DOB,E_CITY,E_SALARY,E_DNO;
EXIT WHEN EMP%NOTFOUND;
DBMS_OUTPUT.PUT_LINE(E_ID || ' ' || E_NAME || ' ' || ' ' || E_DOB || ' ' || E_CITY|| ' ' || ' ' || E_SALARY || ' ' || E_DNO);
END LOOP;
CLOSE EMP;
END;
/
